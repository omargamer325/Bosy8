    function loadCart() {
    try {
    const saved = localStorage.getItem(CONFIG.STORAGE_CART);
    if (saved) {
    const parsed = JSON.parse(saved);
    cart = (parsed || []).map(item => {
    const product = PRODUCTS.find(p => p.id === item.productId);
    return product ? { product, qty: item.qty } : null;
    }).filter(Boolean);
    }
    } catch (e) {
    console.warn('Failed to load cart', e);
    }
    }
    function saveCart() {

    }));
    localStorage.setItem(CONFIG.STORAGE_CART, JSON.stringify(toSave));

    }
    }

    function getCartCount() {
    return cart.reduce((sum, item) => sum + item.qty, 0);
    }
    function getCartSubtotal() {
    return cart.reduce((sum, item) => sum + item.product.price * item.qty, 0);

    function getCartDiscountAmount() {
    if (!appliedCoupon) return 0;
    const subtotal = getCartSubtotal();
    const c = COUPONS[appliedCoupon];

    if (c.type === 'percent') return Math.round(subtotal * c.value / 100);
    return Math.min(c.value, subtotal);
    }
    function getCartTotal() {

    function getFilteredProducts() {

    return list;

    address: (elements.customerAddress && elements.customerAddress.value) || ''

    function loadCustomerDraft() {
    try {
    const data = JSON.parse(localStorage.getItem('baqalati_customer_draft') || '{}');
    if (elements.customerName && data.name) elements.customerName.value = data.name;
    if (elements.customerPhone && data.phone) elements.customerPhone.value = data.phone;
    if (elements.customerAddress && data.address) elements.customerAddress.value = data.address;
    } catch (e) {}
    }
    // حفظ عند الخروج أو كل 2 ثانية عند الكتابة
    if (elements.customerName) elements.customerName.addEventListener('blur', saveCustomerDraft);
    if (elements.customerPhone) elements.customerPhone.addEventListener('blur', saveCustomerDraft);
    if (elements.customerAddress) elements.customerAddress.addEventListener('blur', saveCustomerDraft);
